    name: Deploy App to CentOS via SSH
    runs-on: ubuntu-latest

    steps:
      - name: Checkout Repository
        uses: actions/checkout@v3

      - name: Set up SSH Connection
        run: |
          mkdir -p ~/.ssh
          echo "${{ secrets.SSH_PRIVATE_KEY_249 }}" > ~/.ssh/id_ed25519_249
          chmod 600 ~/.ssh/id_ed25519_249
          ssh-keyscan -p 22 -H 114.130.69.249 >> ~/.ssh/known_hosts

      - name: Deploy to Server
        run: |
          ssh -i ~/.ssh/id_ed25519_249 -p 22 root@114.130.69.249 << 'EOF'
            echo "Connected to CentOS server"

            # Navigate to your project directory

            cd /var/www/html

            # If folder doesn't exist, clone it fresh (use your actual repo)

            if [ ! -d "saas-crm-ihelp/.git" ]; then
              rm -rf saas-crm-ihelp
              git clone -b prod git@github.com:mistersakil/ihelp-sass-crm-20250504.git saas-crm-ihelp
            else
              cd saas-crm-ihelp
              git pull origin prod
            fi

            cd saas-crm-ihelp

            git pull origin prod
