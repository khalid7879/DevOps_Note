pipeline {
    agent {
        node {
            label 'dev'
        }
    }

    stages {
        stage('Clone Repository') {
            steps {
                echo 'Cloning main branch from GitHub...'
                git branch: 'main', url: 'https://github.com/bongodev/flask-student-attendance-app.git'
            }
        }

        stage('Docker Build') {
            steps {
                echo 'Building Docker image...'
                sh '''
                    docker build -t flask-student-app:latest .
                '''
            }
        }

        stage('Deploy with Docker Compose') {
            steps {
                echo 'Redeploying application with Docker Compose...'
                sh '''
                    docker compose down
                    docker compose up -d --build
                '''
            }
        }
    }

    post {
        success {
            echo '✅ Deployment completed successfully!'
        }
        failure {
            echo '❌ Deployment failed. Please review the build logs.'
        }
    }
}
